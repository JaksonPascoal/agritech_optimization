from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression

# Separe as variveis preditoras (X) da varivel alvo (y)
# A varivel alvo  a 'Aplicacao_nitrogenio'
# Use drop() para remover colunas que no sero usadas como preditoras
X = df_encoded.drop(columns=['Data', 'Aplicacao_nitrogenio'])
y = df_encoded['Aplicacao_nitrogenio']

# Divida os dados em conjuntos de treino e teste
# test_size=0.20 significa que 20% dos dados sero usados para teste
# random_state=42 garante que a diviso seja a mesma sempre que voc rodar o cdigo
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.20, random_state=42)

# Exiba o tamanho dos conjuntos para confirmar a diviso
print(f"Tamanho do conjunto de treino (X_train): {X_train.shape}")
print(f"Tamanho do conjunto de teste (X_test): {X_test.shape}")

# Inicialize o modelo de Regresso Linear
model = LinearRegression()

# Treine o modelo com os dados de treino
model.fit(X_train, y_train)

print("\nModelo treinado com sucesso!")
